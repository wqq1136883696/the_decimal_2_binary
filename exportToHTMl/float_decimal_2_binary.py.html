12   from decimal import Decimal
13   
14   
15   def divide_float(f_number):
16       """ 
17           分离整数和小数，返回符号位，整数部分，和小数部分 
18       """
19       # 转换成正数
20       sign = 0
21       if f_number < 0:
22           sign = 1
23           f_number = -1 * f_number
24   
25       # 整数部分
26       integer = int(f_number)
27   
28       # 小数部分
29       decimal_part = Decimal(str(f_number)) - integer
30   
31       return sign, integer, decimal_part
32   
33   
34   def int2bin(integer):
35       """ 
36           把整数部分转化成二进制，整数部分不为O 
37           返回转换后的整数部分 
38       """
39       integer_covert = str(bin(integer))
40       return integer_covert[2:]
41   
42   
43   def decimal_part2bin(decimal_part):
44       """把小数部分转化成二进制,小数部分不为O"""
45       decimal_convert = ""
46   
47       for i in range(23):
48           result = decimal_part * 2
49           # print(result)
50           decimal_convert += str(int(result))
51           decimal_part = result - int(result)
52   
53           if result == 1:
54               break
55   
56       return decimal_convert
57   
58   
59   def get_bin(f_number):
60       """得到完整的二进制表示"""
61       sign, integer, decimal = divide_float(f_number)
62       # 判断如果整数部分和小数部分都为0，则全为0
63       if integer == 0 and decimal == 0:
64           f_number_bin = "0" * 32
65       else:
66           # 如果整数部分为零，则显示为0
67           if integer != 0:
68               integer_bin = int2bin(integer)
69           else:
70               integer_bin = "0"
71   
72           # 如果小数部分为零，则不显示
73           if decimal != 0:
74               decimal_bin = decimal_part2bin(decimal)
75               f_number_bin = integer_bin + "." + decimal_bin
76           else:
77               f_number_bin = integer_bin
78   
79       return f_number_bin

83   def main():
