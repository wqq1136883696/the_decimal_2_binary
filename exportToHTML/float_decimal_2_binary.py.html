12   from decimal import Decimal
13   
14   
15   def divide_float(f_number):
16       """分离整数和小数"""
17       # 转换成正数
18       sign = 0
19       if f_number < 0:
20           sign = 1
21           f_number = -1 * f_number
22   
23       # 整数部分
24       integer = int(f_number)
25   
26       # 小数部分
27       decimal_part = Decimal(str(f_number)) - integer
28   
29       return sign, integer, decimal_part
30   
31   
32   def int2bin(integer):
33       """把整数部分转化成二进制，整数部分不为O"""
34       integer_covert = str(bin(integer))
35       return integer_covert[2:]
36   
37   
38   def decimal_part2bin(decimal_part):
39       """把小数部分转化成二进制,小数部分不为O"""
40       decimal_convert = ""
41   
42       for i in range(23):
43           result = decimal_part * 2
44           # print(result)
45           decimal_convert += str(int(result))
46           decimal_part = result - int(result)
47   
48           if result == 1:
49               break
50   
51       return decimal_convert
52   
53   
54   def get_bin(f_number):
55       """得到完整的二进制表示"""
56       sign, integer, decimal = divide_float(f_number)
57       # 判断如果整数部分和小数部分都为0，则全为0
58       if integer == 0 and decimal == 0:
59           f_number_bin = "0" * 32
60       else:
61           # 如果整数部分为零，则显示为0
62           if integer != 0:
63               integer_bin = int2bin(integer)
64           else:
65               integer_bin = "0"
66   
67           # 如果小数部分为零，则不显示
68           if decimal != 0:
69               decimal_bin = decimal_part2bin(decimal)
70               f_number_bin = integer_bin + "." + decimal_bin
71           else:
72               f_number_bin = integer_bin
73   
74       return f_number_bin
75   
76   
77   def main():
78       """主程序的执行"""
79       while True:
80           print("输入q退出程序")
81           input_str = input("请输入一个十进制数：")
82           if input_str == "q":
83               break
84           try:
85               f_number = Decimal(input_str)
86               bin_number = get_bin(f_number)
87               print(bin_number)
88           except:
89               print("您的输入有误，请输入一个数字！")
90   
91   
92   if __name__ == '__main__':
93       main()
