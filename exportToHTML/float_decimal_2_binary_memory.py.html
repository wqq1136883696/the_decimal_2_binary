14   from decimal import Decimal
15   
16   BIAS = 127  # 指数位的偏移量
17   
18   
19   def divide_float(f_number):
20       """分离整数和小数"""
21       # 转换成正数
22       sign = "0"
23       if f_number < 0:
24           sign = "1"
25           f_number = -1 * f_number
26   
27       # 整数部分
28       integer = int(f_number)
29   
30       # 小数部分
31       decimal_part = Decimal(str(f_number)) - integer
32   
33       return sign, integer, decimal_part
34   
35   
36   def int2bin(integer):
37       """把整数部分转化成二进制，整数部分不为O"""
38       integer_covert = str(bin(integer))
39       return integer_covert[2:]
40   
41   
42   def decimal_part2bin(decimal_part):
43       """把小数部分转化成二进制,小数部分不为O"""
44       decimal_convert = ""
45   
46       for i in range(32):
47           result = decimal_part * 2
48           # print(result)
49           decimal_convert += str(int(result))
50           decimal_part = result - int(result)
51   
52           if result == 1:
53               break
54   
55       return decimal_convert
56   
57   
58   def get_bin(f_number):
59       """得到完整的二进制表示"""
60       sign, integer, decimal = divide_float(f_number)
61       # 判断如果整数部分和小数部分都为0，则全为0
62       if integer == 0 and decimal == 0:
63           f_number_bin = "0"
64       else:
65           # 如果整数部分为零，则显示为0
66           if integer != 0:
67               integer_bin = int2bin(integer)
68           else:
69               integer_bin = "0"
70   
71           # 如果小数部分为零，则不显示
72           if decimal != 0:
73               decimal_bin = decimal_part2bin(decimal)
74               f_number_bin = integer_bin + "." + decimal_bin
75           else:
76               f_number_bin = integer_bin
77   
78       return f_number_bin, sign
79   
80   
81   def science_display(f_number_bin):
82       """把二进制显示为科学计数表示,返回指数部分和小数部分的值 
83           例如： 1.01*2**3，默认整数位为1 
84       """
85       str_part = f_number_bin.split('.')
86       integer_bin = str_part[0]
87       decimal_bin = ""
88       if len(str_part) == 2:
89           decimal_bin = str_part[1]
90   
91       if integer_bin != "0":
92           # print(integer_bin)
93           integer_length = len(integer_bin)
94           exponent = integer_length - 1  # 科学计数法的指数部分
95           # print(exponent)
96           decimal_part1 = integer_bin[1:]  # 第一位为1，去剩下的作为小数部分的一部分
97           decimal_part = decimal_part1 + decimal_bin  # 与原有的小数部分组合成一个新的小数部分
98           # print(decimal_part)
99       else:
100          # print(decimal_bin)
101          position = decimal_bin.find("1") + 1  # 小数部分第一个非0的位置
102          exponent = -position  # 科学计数法的指数部分
103          # print(exponent)
104          decimal_part = decimal_bin[position:]
105          # print(decimal_part)
106  
107      return exponent, decimal_part

110  def display_in_memory(f_number_bin, sign, exponent, decimal_part):
111      """在计算机内存中的显示"""
112  
113      if f_number_bin == "0":
114          f_number_memory = "0" * 32
115      else:
116          # 把指数位显示出来
117          exponent = exponent + BIAS
118          exponent_bin = bin(exponent)[2:]
119          # print(exponent_bin)
120          # 如果不足8位，在前面加0
121          exponent_length = len(exponent_bin)
122          add_number = 8 - exponent_length
123          exponent_bin_memory = "0" * add_number + exponent_bin
124          # print(exponent_bin_memory)
125  
126          # 把小数位显示出来，默认整数位为1,如果不足23位，在后面加0补足
127          if len(decimal_part) >= 23:
128              decimal_part_memory = decimal_part[:23]
129          else:
130              add_number_decimal = 23 - len(decimal_part)
131              decimal_part_memory = decimal_part + "0" * add_number_decimal
132          # print(decimal_part_memory)
133  
134          f_number_memory = sign + exponent_bin_memory + decimal_part_memory
135  
136      return f_number_memory

154                    % (input_str, bin_number, f_number_memory))
155          except Exception:
