14   from decimal import Decimal
15   
16   BIAS = 127  # 指数位的偏移量
17   
18   
19   def divide_float(f_number):
20       """分离整数和小数, 返回符号位，整数部分，和小数部分"""
21   
22       # 转换成正数
23       sign = "0"
24       if f_number < 0:
25           sign = "1"
26           f_number = -1 * f_number
27   
28       # 整数部分
29       integer = int(f_number)
30   
31       # 小数部分
32       decimal_part = Decimal(str(f_number)) - integer
33   
34       return sign, integer, decimal_part
35   
36   
37   def int2bin(integer):
38       """把整数部分转化成二进制，整数部分不为O 
39       返回转换后的整数部分 
40       """
41       integer_covert = str(bin(integer))
42       return integer_covert[2:]
43   
44   
45   def decimal_part2bin(decimal_part):
46       """把小数部分转化成二进制,小数部分不为O"""
47       decimal_convert = ""
48   
49       for i in range(32):
50           result = decimal_part * 2
51           # print(result)
52           decimal_convert += str(int(result))
53           decimal_part = result - int(result)
54   
55           if result == 1:
56               break
57   
58       return decimal_convert
59   
60   
61   def get_bin(f_number):
62       """得到完整的二进制表示"""
63       sign, integer, decimal = divide_float(f_number)
64       # 判断如果整数部分和小数部分都为0，则全为0
65       if integer == 0 and decimal == 0:
66           f_number_bin = "0"
67       else:
68           # 如果整数部分为零，则显示为0
69           if integer != 0:
70               integer_bin = int2bin(integer)
71           else:
72               integer_bin = "0"
73   
74           # 如果小数部分为零，则不显示
75           if decimal != 0:
76               decimal_bin = decimal_part2bin(decimal)
77               f_number_bin = integer_bin + "." + decimal_bin
78           else:
79               f_number_bin = integer_bin
80   
81       return f_number_bin, sign
82

84   def science_display(f_number_bin):
85       """把二进制显示为科学计数表示,返回指数部分和小数部分的值 
86           例如： 1.01*2**3，默认整数位为1 
87       """
88       str_part = f_number_bin.split('.')
89       integer_bin = str_part[0]
90       decimal_bin = ""
91       if len(str_part) == 2:
92           decimal_bin = str_part[1]
93   
94       if integer_bin != "0":
95           # print(integer_bin)
96           integer_length = len(integer_bin)
97           exponent = integer_length - 1  # 科学计数法的指数部分
98           # print(exponent)
99           decimal_part1 = integer_bin[1:]  # 第一位为1，去剩下的作为小数部分的一部分
100          decimal_part = decimal_part1 + decimal_bin  # 与原有的小数部分组合成一个新的小数部分
101          # print(decimal_part)
102      else:
103          # print(decimal_bin)
104          position = decimal_bin.find("1") + 1  # 小数部分第一个非0的位置
105          exponent = -position  # 科学计数法的指数部分
106          # print(exponent)
107          decimal_part = decimal_bin[position:]
108          # print(decimal_part)
109  
110      return exponent, decimal_part

113  def display_in_memory(f_number_bin, sign, exponent, decimal_part):
114      """在计算机内存中的显示"""
115  
116      if f_number_bin == "0":
117          f_number_memory = "0" * 32
118      else:
119          # 把指数位显示出来
120          exponent = exponent + BIAS
121          exponent_bin = bin(exponent)[2:]
122          # print(exponent_bin)
123          # 如果不足8位，在前面加0
124          exponent_length = len(exponent_bin)
125          add_number = 8 - exponent_length
126          exponent_bin_memory = "0" * add_number + exponent_bin
127          # print(exponent_bin_memory)
128  
129          # 把小数位显示出来，默认整数位为1,如果不足23位，在后面加0补足
130          if len(decimal_part) >= 23:
131              decimal_part_memory = decimal_part[:23]
132          else:
133              add_number_decimal = 23 - len(decimal_part)
134              decimal_part_memory = decimal_part + "0" * add_number_decimal
135          # print(decimal_part_memory)
136  
137          f_number_memory = sign + exponent_bin_memory + decimal_part_memory
138  
139      return f_number_memory

157                    % (input_str, bin_number, f_number_memory))
158          except Exception:
